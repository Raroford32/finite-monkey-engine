options:
  -h, --help            show this help message and exit
  --path PATH           Path to the project directory for exploit discovery
                        analysis
  --project-id PROJECT_ID
                        Project ID for analysis (auto-generated from path if
                        not provided)
  --mode {test,direct_excel,exploit_discovery}
                        Operation mode (default: exploit_discovery)
  --output OUTPUT       Output file path for the analysis report
  --generate-report     Generate report from existing analysis data
  --scan-mode {PURE_SCAN,COMMON_PROJECT,CHECKLIST,COMMON_PROJECT_FINE_GRAINED}
                        Vulnerability scanning mode (default:
                        COMMON_PROJECT_FINE_GRAINED)

Examples:
  # Analyze a project by providing just the path
  python src/main.py --path /path/to/your/project
  
  # Generate report from existing scan results
  python src/main.py --generate-report --project-id my_project
  
  # Direct Excel generation mode
  python src/main.py --mode direct_excel --project-id pebble
